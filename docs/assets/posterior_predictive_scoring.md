def compute_anomaly_score(y_new, entity_idx, posterior_samples):
    """
    y_new: New observation count
    entity_idx: Which entity (0, 1, ..., N-1)
    posterior_samples: Dict with keys ['theta', 'phi']
                       theta.shape = (n_samples, n_entities)
                       phi.shape = (n_samples,)
    """
    # Extract samples for this entity
    theta_samples = posterior_samples['theta'][:, entity_idx]
    phi_samples = posterior_samples['phi']
    
    # Compute P(y_new | each posterior sample)
    log_probs = []
    for theta, phi in zip(theta_samples, phi_samples):
        # Negative Binomial log PMF
        log_p = nbinom_logpmf(y_new, mu=theta, alpha=phi)
        log_probs.append(log_p)
    
    # Average in log space (log-sum-exp trick for stability)
    avg_log_prob = logsumexp(log_probs) - log(len(log_probs))
    
    # Anomaly score
    score = -avg_log_prob
    
    return score
